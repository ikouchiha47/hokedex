Toggle handler:
```typescript
function handleToggle() {
  const nextOpen = !open;
  setOpen(nextOpen);

  const cfg = { duration: ANIMATION_DURATION, easing: EASING };

  if (nextOpen) {
    contactX.value = withTiming(CONTACT_TARGET.x, cfg);
    contactY.value = withTiming(CONTACT_TARGET.y, cfg);
    contactOpacity.value = withTiming(1, cfg);

    micX.value = withTiming(MIC_TARGET.x, cfg);
    micY.value = withTiming(MIC_TARGET.y, cfg);
    micOpacity.value = withTiming(1, cfg);

    cameraX.value = withTiming(CAMERA_TARGET.x, cfg);
    cameraY.value = withTiming(CAMERA_TARGET.y, cfg);
    cameraOpacity.value = withTiming(1, cfg);
  } else {
    contactX.value = withTiming(0, cfg);
    contactY.value = withTiming(0, cfg);
    contactOpacity.value = withTiming(0, cfg);

    micX.value = withTiming(0, cfg);
    micY.value = withTiming(0, cfg);
    micOpacity.value = withTiming(0, cfg);

    cameraX.value = withTiming(0, cfg);
    cameraY.value = withTiming(0, cfg);
    cameraOpacity.value = withTiming(0, cfg);
  }
}
```
