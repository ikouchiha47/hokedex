    This screen is a thin orchestrator. It does NOT call HokedexML directly. It calls HokedexMediaModule for acquisition, then passes the imageUri to `cameraCaptureFlow` (implemented in Plan 03-02). For Plan 03-01, stub the flow call:

    ```typescript
    import React, { useCallback, useState } from 'react';
    import { View, ActivityIndicator, ToastAndroid, StyleSheet } from 'react-native';
    import { useNavigation } from '@react-navigation/native';
    import { HokedexMediaModule } from '../native/HokedexMediaModule'; // check actual import path
    import FacePickerModal from '../components/FacePickerModal';
    import PersonConfirmModal, { PersonConfirmResult } from '../components/PersonConfirmModal';

    // Flow will be wired in Plan 03-03
    type ScreenState =
      | { stage: 'idle' }
      | { stage: 'loading' }
      | { stage: 'face_picker'; imageUri: string }
      | { stage: 'person_confirm'; imageUri: string; faceIndex: number }
      | { stage: 'done' };

    export function CameraCaptureScreen() {
      const navigation = useNavigation();
      const [state, setState] = useState<ScreenState>({ stage: 'idle' });

      const handleCapture = useCallback(async () => {
        setState({ stage: 'loading' });
        const result = await HokedexMediaModule.capturePhoto();
        if (!result || !result.uri) { setState({ stage: 'idle' }); return; }
        setState({ stage: 'face_picker', imageUri: result.uri });
      }, []);

      const handleGallery = useCallback(async () => {
        setState({ stage: 'loading' });
        const result = await HokedexMediaModule.pickImageFromGallery();
        if (!result || !result.uri) { setState({ stage: 'idle' }); return; }
        setState({ stage: 'face_picker', imageUri: result.uri });
      }, []);

      // Plan 03-03 will replace this stub with real flow
      const handleFacesConfirmed = useCallback((_faces: unknown) => {
        ToastAndroid.show('Moment saved (stub)', ToastAndroid.SHORT);
        navigation.goBack();
      }, [navigation]);

      // On mount â€” immediately open camera (primary path per R-CAM-01)
      React.useEffect(() => { handleCapture(); }, [handleCapture]);

      if (state.stage === 'loading') {
        return <View style={styles.center}><ActivityIndicator size="large" /></View>;
      }

      return (
        <View style={styles.container}>
          {state.stage === 'face_picker' && (
            <FacePickerModal
              imageUri={state.imageUri}
              onConfirm={handleFacesConfirmed}
              onDismiss={() => navigation.goBack()}
            />
          )}
        </View>
      );
    }

    const styles = StyleSheet.create({
      container: { flex: 1, backgroundColor: '#000' },
      center: { flex: 1, alignItems: 'center', justifyContent: 'center' },
    });
    ```
