    ): Promise<DetectResult> {
      const result = await modules.HokedexML.detect(imageUri);
      // Handle all 4 DetectionResult variants from ml.ts (NO_FACE, SINGLE, MULTI_SUBJECT, ERROR)
      // Check actual variant names in src/types/ml.ts before writing this switch
      // No-face and error both return { kind: 'no_face' }

    }

    /**
     * R-CAM-05: For one confirmed face crop, run embedding and match against DB.
     * categoryId scopes the vector search to the correct embedding namespace.
     */
    export async function runEmbedAndMatch(
      modules: CameraFlowModules,
      db: DB,
      imageUri: string,
      crop: BoundingBox,
      categoryId: string,
    ): Promise<MatchResult> {
      const vector = await embedCrop(
        { HokedexML: modules.HokedexML },
        { imageUri, selectedCrop: crop, categoryId },
      );
      const queryBuffer = float32ToBuffer(vector);
      const matches = await searchEmbeddingsByVector(db, queryBuffer, categoryId);
      const best = matches[0];
      if (best && best.best_score >= SIMILARITY_THRESHOLD) {
        // Return match — check actual field names on the matches result from embeddings.ts
        // before hardcoding entry_id / name field names
        return { kind: 'match', entryId: best.entry_id, candidateName: best.name ?? '', score: best.best_score };
      }
      return { kind: 'no_match' };
    }
