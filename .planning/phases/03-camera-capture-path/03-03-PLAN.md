    ```typescript
    async capture(input: CaptureInput & { source?: string }): Promise<Result<CaptureResult>> {
      try {
        let momentId!: string;
        withTransaction(this.db, tx => {
          momentId = insertMoment(tx, input.note, input.occurredAt, input.placeId ?? null);
          // Set source if provided — uses UPDATE because insertMoment doesn't accept source yet.
          // ALTERNATIVE: read insertMoment's SQL signature first. If INSERT_MOMENT SQL already

          // Choose whichever keeps SQL in .sql files per CLAUDE.md conventions.
          input.entryIds.forEach(entryId => insertMomentPerson(tx, momentId, entryId));
        });
        return { ok: true, value: { momentId } };
      } catch (err) {
        return { ok: false, error: String(err) };
      }
    }
    ```

    deciding approach. If `INSERT_MOMENT` already has a `source` placeholder (it does NOT

    path per CLAUDE.md ("no raw SQL strings in TypeScript"). If the SQL file is not writable
    in this plan's scope, use the UPDATE fallback as a temporary measure and add a TODO comment.

    PREFERRED approach (follow CLAUDE.md): update `src/db/sql/queries/moments.sql` INSERT_MOMENT
